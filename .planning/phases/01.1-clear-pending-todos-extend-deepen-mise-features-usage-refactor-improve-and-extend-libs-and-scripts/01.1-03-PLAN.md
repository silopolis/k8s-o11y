## Context

**Locked Decisions (NON-NEGOTIABLE):**

1. **Script/Libs Structure:** Top-level directories - `scripts/` for task scripts, `lib/` for shared libraries

  <action>Create `lib/kubernetes.sh` with domain-specific Kubernetes helper functions:

  ```bash
  #!/usr/bin/env bash
  # Kubernetes-specific helper functions
  # Source this file in scripts: source "${PROJECT_ROOT}/lib/kubernetes.sh"
  
  # Get cluster name from current context
  k8s_get_cluster_name() {
      kubectl config current-context 2>/dev/null || echo "unknown"
  }
  
  # Get cluster endpoint
  k8s_get_cluster_endpoint() {
      kubectl config view --minify -o jsonpath='{.clusters[0].cluster.server}' 2>/dev/null || echo "unknown"
  }
  
  # Get cluster status (Ready/NotReady node count)
  k8s_get_cluster_status() {
      local ready=$(kubectl get nodes -o jsonpath='{.items[*].status.conditions[?(@.type=="Ready")].status}' 2>/dev/null | grep -o "True" | wc -l)
      local total=$(kubectl get nodes --no-headers 2>/dev/null | wc -l)
      echo "${ready}/${total} nodes ready"
  }
  
  # Get monitoring namespace pod status
  k8s_get_monitoring_pods() {
      kubectl get pods -n monitoring --no-headers 2>/dev/null | wc -l
  }
  
  # Get Prometheus URL (internal)
  k8s_get_prometheus_url() {
      echo "http://kube-prometheus-stack-prometheus.monitoring.svc.cluster.local:9090"
  }
  
  # Get Grafana NodePort URL
  k8s_get_grafana_url() {
      local node_ip=$(kubectl get nodes -o jsonpath='{.items[0].status.addresses[?(@.type=="InternalIP")].address}' 2>/dev/null)
      echo "http://${node_ip}:30030"
  }
  
  # Get Alertmanager URL (internal)
  k8s_get_alertmanager_url() {
      echo "http://kube-prometheus-stack-alertmanager.monitoring.svc.cluster.local:9093"
  }
  
  # Check if Prometheus CRDs are installed
  k8s_check_prometheus_crds() {
      kubectl get crd prometheuses.monitoring.coreos.com >/dev/null 2>&1 && echo "installed" || echo "missing"
  }
  
  # Get pod status summary for a namespace
  k8s_get_pod_summary() {
      local namespace="${1:-monitoring}"
      kubectl get pods -n "$namespace" -o custom-columns='NAME:.metadata.name,STATUS:.status.phase,READY:.status.containerStatuses[*].ready' --no-headers 2>/dev/null
  }
  ```

  All functions should handle errors gracefully and return "unknown" or sensible defaults when kubectl can't connect. Document that this file should be sourced, not executed directly.</action>

<task type="auto" id="3">
  <title>Create lib/versions.sh with version getters</title>
  <files>lib/versions.sh</files>
  <action>Create `lib/versions.sh` with version getter functions for all tools and cluster:

  ```bash
  #!/usr/bin/env bash
  # Version getter functions for tools and cluster
  # Source this file in scripts: source "${PROJECT_ROOT}/lib/versions.sh"
  
  # Get Talos version
  version_talos() {
      talosctl version --short 2>/dev/null || echo "unknown"
  }
  
  # Get Talos node versions
  version_talos_nodes() {
      talosctl get nodes -o json 2>/dev/null | jq -r '.[].spec.version' 2>/dev/null || echo "unknown"
  }
  
  # Get kubectl version
  version_kubectl() {
      kubectl version --client --short 2>/dev/null || echo "unknown"
  }
  
  # Get kubectl server version
  version_kubectl_server() {
      kubectl version --short 2>/dev/null | grep Server | awk '{print $3}' || echo "unknown"
  }
  
  # Get helm version
  version_helm() {
      helm version --short 2>/dev/null || echo "unknown"
  }
  
  # Get helmfile version
  version_helmfile() {
      helmfile version --short 2>/dev/null || echo "unknown"
  }
  
  # Get kube-prometheus-stack chart version
  version_kps_chart() {
      helm list -n monitoring -o json 2>/dev/null | jq -r '.[] | select(.name=="kube-prometheus-stack") | .chart' 2>/dev/null || echo "unknown"
  }
  
  # Get Prometheus operator version
  version_prometheus_operator() {
      kubectl get deployment -n monitoring kube-prometheus-stack-operator -o jsonpath='{.spec.template.spec.containers[0].image}' 2>/dev/null | cut -d: -f2 || echo "unknown"
  }
  
  # Get Grafana version
  version_grafana() {
      kubectl get deployment -n monitoring kube-prometheus-stack-grafana -o jsonpath='{.spec.template.spec.containers[0].image}' 2>/dev/null | cut -d: -f2 || echo "unknown"
  }
  
  # Get all versions as formatted report
  version_report() {
      echo "=== Tool Versions ==="
      echo "Talos: $(version_talos)"
      echo "kubectl: $(version_kubectl)"
      echo "Helm: $(version_helm)"
      echo "Helmfile: $(version_helmfile)"
      echo ""
      echo "=== Cluster Versions ==="
      echo "Kubernetes Server: $(version_kubectl_server)"
      echo "KPS Chart: $(version_kps_chart)"
      echo "Prometheus Operator: $(version_prometheus_operator)"
      echo "Grafana: $(version_grafana)"
  }
  ```

  Functions should gracefully handle errors (tools not installed, cluster not accessible) by returning "unknown". The version_report function provides a comprehensive overview.</action>
  <verify>Run `shellcheck -x lib/versions.sh` to verify it passes linting; test sourcing and calling one function to verify it works</verify>
  <done>lib/versions.sh exists with all version getters and passes shellcheck</done>
