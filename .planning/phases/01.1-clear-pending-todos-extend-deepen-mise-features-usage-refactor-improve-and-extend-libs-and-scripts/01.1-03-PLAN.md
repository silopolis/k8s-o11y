    #!/usr/bin/env bash
    # Common/shared utility functions for all scripts
    # Source this file at the start of scripts: source lib/common.sh
    
    # Colors for output (only if terminal supports it)
    if [[ -t 1 ]]; then
      readonly RED='\033[0;31m'
      readonly GREEN='\033[0;32m'
      readonly YELLOW='\033[1;33m'
      readonly BLUE='\033[0;34m'
      readonly NC='\033[0m' # No Color
    else
      readonly RED=''
      readonly GREEN=''
      readonly YELLOW=''
      readonly BLUE=''
      readonly NC=''
    fi
    
    # Print functions
    info() { echo -e "${BLUE}[INFO]${NC} $*"; }
    success() { echo -e "${GREEN}[SUCCESS]${NC} $*"; }
    warn() { echo -e "${YELLOW}[WARN]${NC} $*"; }
    error() { echo -e "${RED}[ERROR]${NC} $*" >&2; }
    
    # Error handling
    die() { error "$*"; exit 1; }
    check_command() {
      local cmd="$1"
      command -v "$cmd" >/dev/null 2>&1 || die "Required command '$cmd' not found"
    }
    
    # Utility functions
    confirm() {
      local prompt="${1:-Continue?}"
      read -rp "$prompt [y/N]: " response
      [[ "$response" =~ ^[Yy]$ ]]
    }
    
    # Version comparison
    version_ge() {
      printf '%s\n%s\n' "$2" "$1" | sort -V -C
    }
    ```

  <title>Create lib/kubernetes.sh with getter functions</title>
  <files>lib/kubernetes.sh</files>
  <action>
    Create Kubernetes-specific utilities at `lib/kubernetes.sh`:
    ```bash
    #!/usr/bin/env bash
    # Kubernetes and cluster-related utility functions
    # Depends on: lib/common.sh (source both)
    
    source "$(dirname "${BASH_SOURCE[0]}")/common.sh"
    
    # Get Kubernetes version from cluster
    get_k8s_version() {
      kubectl version --json 2>/dev/null | jq -r '.serverVersion.gitVersion' || echo "unknown"
    }
    
    # Get Talos version from cluster nodes
    get_talos_version() {
      kubectl get nodes -o json 2>/dev/null | \
        jq -r '.items[0].status.nodeInfo.osImage' | \
        grep -oP 'v\K[0-9.]+' || echo "unknown"
    }
    
    # Get number of ready nodes
    get_ready_nodes_count() {
      kubectl get nodes 2>/dev/null | grep -c 'Ready' || echo "0"
    }
    
    # Get total nodes count
    get_total_nodes_count() {
      kubectl get nodes --no-headers 2>/dev/null | wc -l || echo "0"
    }
    
    # Get cluster name from current context
    get_cluster_name() {
      kubectl config current-context 2>/dev/null || echo "unknown"
    }
    
    # Get Prometheus version if installed
    get_prometheus_version() {
      kubectl get deployment -n monitoring prometheus-kube-prometheus-stack-prometheus -o json 2>/dev/null | \
        jq -r '.spec.template.spec.containers[0].image' | \
        cut -d: -f2 || echo "not-installed"
    }
    
    # Get Grafana version if installed
    get_grafana_version() {
      kubectl get deployment -n monitoring prometheus-grafana -o json 2>/dev/null | \
        jq -r '.spec.template.spec.containers[0].image' | \
        cut -d: -f2 || echo "not-installed"
    }
    
    # Get all versions as JSON
    get_all_versions_json() {
      jq -n \
        --arg k8s "$(get_k8s_version)" \
        --arg talos "$(get_talos_version)" \
        --arg cluster "$(get_cluster_name)" \
        --arg nodes "$(get_total_nodes_count)" \
        --arg ready "$(get_ready_nodes_count)" \
        --arg prometheus "$(get_prometheus_version)" \
        --arg grafana "$(get_grafana_version)" \
        '{
          kubernetes: $k8s,
          talos: $talos,
          cluster_name: $cluster,
          total_nodes: $nodes,
          ready_nodes: $ready,
          prometheus: $prometheus,
          grafana: $grafana
        }'
    }
    
    # Print formatted version info
    print_versions() {
      info "Cluster Information:"
      echo "  Kubernetes: $(get_k8s_version)"
      echo "  Talos: $(get_talos_version)"
      echo "  Cluster: $(get_cluster_name)"
      echo "  Nodes: $(get_ready_nodes_count)/$(get_total_nodes_count) ready"
      echo "  Prometheus: $(get_prometheus_version)"
      echo "  Grafana: $(get_grafana_version)"
    }
    
    # Get tool versions from mise
    get_tool_version() {
      local tool="$1"
      mise exec -- "$tool" --version 2>/dev/null | head -1 || echo "not-installed"
    }
    ```
    This provides getter functions for cluster and tool versions.

    Create thin wrapper script at `scripts/get-cluster-info.sh`:
    ```bash
    #!/usr/bin/env bash
    # Thin wrapper script that calls lib functions
    # Usage: ./scripts/get-cluster-info.sh [json|pretty]
    
    set -euo pipefail
    
    # Get script directory
    SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
    cd "${SCRIPT_DIR}/.."
    
    # Source libraries
    source lib/common.sh
    source lib/kubernetes.sh
    
    # Main execution
    main() {
      local format="${1:-pretty}"
      
      info "Retrieving cluster information..."
      
      case "$format" in
        json)
          get_all_versions_json
          ;;
        pretty|text|*)
          print_versions
          ;;
      esac
      
      success "Done"
    }
    
    main "$@"

    # Setup Guide
    
    Initial setup for new team members working with this project.
    
    ## Prerequisites
    
    - [mise](https://mise.jdx.dev/) installed
    - kubectl configured with cluster access
    - Git configured with your credentials
    
    ## 1. Install Tools
    
    Install all required tools via mise:
    
    ```bash
    mise install
    ```
    
    This installs: kubectl, helm, helmfile, talosctl, sops, age, pre-commit, markdownlint-cli2, shellcheck
    
    ## 2. Get the Age Private Key
    
    **IMPORTANT:** The project uses SOPS with age encryption for secrets management. You need the age private key to decrypt secrets.
    
    ### For New Team Members
    
    The age private key is distributed securely via the team's password manager:
    
    1. Access the shared vault in **1Password** (or Bitwarden)
    2. Find the item named "Project Age Key"
    3. Copy the private key content (starts with `AGE-SECRET-KEY-`)
    4. Create the key file:
       ```bash
       mkdir -p ~/.config/mise/
       # Paste the key into this file
       cat > ~/.config/mise/age.txt << 'EOF'
       # created: 2026-03-12T00:00:00+01:00
       # public key: age1...
       AGE-SECRET-KEY-...
       EOF
       chmod 600 ~/.config/mise/age.txt
       ```
    
    ### Key Security
    
    - **NEVER commit the private key**
    - **NEVER share the private key via chat/email**
    - The private key is in `~/.config/mise/age.txt` (already in .gitignore)
    - Backup: Store a copy in your personal password manager
    - If lost: Contact team lead to generate new key and re-encrypt secrets
    
    ## 3. Verify Secrets Access
    
    Test that you can decrypt secrets:
    
    ```bash
    # This should show decrypted values
    mise env | grep GRAFANA
    
    # Or directly with sops
    sops decrypt .env.json
    ```
    
    ## 4. Install Pre-commit Hooks
    
    Set up git hooks for code quality:
    
    ```bash
    mise exec -- pre-commit install
    ```
    
    ## 5. Verify Setup
    
    Run the verification tasks:
    
    ```bash
    # Verify infrastructure (Wave 1)
    mise run verify:wave1
    
    # Verify quality gates (Wave 2)
    mise run verify:wave2
    
    # Check cluster access
    kubectl get nodes
    ```
    
    ## 6. Common Tasks
    
    ### Edit Secrets
    
    ```bash
    sops edit .env.json
    ```
    
    ### Add New Secret
    
    ```bash
    sops set .env.json '"NEW_KEY": "value"'
    ```
    
    ### View Cluster Info
    
    ```bash
    mise run utils:versions
    ```
    
    ## Troubleshooting
    
    ### "age: no identity matched any of the recipients"
    
    The age private key is missing or wrong. Verify:
    1. File exists: `cat ~/.config/mise/age.txt`
    2. Contains private key (starts with `AGE-SECRET-KEY-`)
    3. Permissions are 600: `chmod 600 ~/.config/mise/age.txt`
    
    ### "command not found: mise"
    
    mise is not installed or not in PATH. Follow mise installation guide.
    
    ### Pre-commit fails
    
    Run manually to see details:
    ```bash
    mise exec -- pre-commit run --all-files
    ```
