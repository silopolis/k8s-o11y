<task type="auto" id="2">
  <title>Create lib/lint.sh shared utilities</title>
  <files>lib/lint.sh</files>
  <action>
    Create shared linting utilities at `lib/lint.sh`:
    ```bash
    #!/usr/bin/env bash
    # Shared linting utilities for scripts and tasks
    
    # Run markdownlint on specified files or all markdown
    lint_markdown() {
      local files=("${@:-**/*.md}")
      mise exec -- markdownlint-cli2 "${files[@]}"
    }
    
    # Run markdownlint with auto-fix
    lint_markdown_fix() {
      local files=("${@:-**/*.md}")
      mise exec -- markdownlint-cli2 "${files[@]}" --fix
    }
    
    # Run shellcheck on specified shell scripts
    lint_shell() {
      local files=("${@}")
      if [[ ${#files[@]} -eq 0 ]]; then
        # Find all .sh files excluding node_modules and .git
        mapfile -t files < <(find . -name "*.sh" -not -path "./node_modules/*" -not -path "./.git/*")
      fi
      mise exec -- shellcheck "${files[@]}"
    }
    
    # Run shellcheck with severity warning
    lint_shell_strict() {
      local files=("${@}")
      if [[ ${#files[@]} -eq 0 ]]; then
        mapfile -t files < <(find . -name "*.sh" -not -path "./node_modules/*" -not -path "./.git/*")
      fi
      mise exec -- shellcheck --severity=warning "${files[@]}"
    }
    
    # Check if tools are available
    lint_check_tools() {
      command -v mise >/dev/null 2>&1 || { echo "mise not found"; return 1; }
      mise exec -- markdownlint-cli2 --version >/dev/null 2>&1 || { echo "markdownlint-cli2 not available via mise"; return 1; }
      mise exec -- shellcheck --version >/dev/null 2>&1 || { echo "shellcheck not available via mise"; return 1; }
      echo "All linting tools available"
      return 0
    }
    ```
    This library provides reusable lint functions for tasks and scripts.
  </action>
  <verify>Run `source lib/lint.sh && lint_check_tools` and verify it reports tools available</verify>
  <done>lib/lint.sh exists with lint_markdown, lint_shell, and lint_check_tools functions</done>

<task type="auto" id="4">
  <title>Create .pre-commit-config.yaml with mise exec pattern</title>
  <files>.pre-commit-config.yaml</files>
  <action>
    Create pre-commit configuration using `mise exec --` pattern for version consistency (per re-research insight):
    ```yaml
    # Pre-commit hooks configuration
    # Uses mise exec pattern to ensure version consistency with mise.toml
    
    repos:
      - repo: local
        hooks:
          # Trailing whitespace - auto-fix then pass
          - id: trailing-whitespace
            name: Trim trailing whitespace
            entry: mise exec -- trailing-whitespace-fixer
            language: system
            types: [file]
            stages: [pre-commit]
          
          # End of file fixer - auto-fix then pass
          - id: end-of-file-fixer
            name: Fix end of files
            entry: mise exec -- end-of-file-fixer
            language: system
            types: [file]
            stages: [pre-commit]
          
          # Markdown linting - WARN only (don't block)
          - id: markdownlint
            name: Markdown linting
            entry: mise exec -- markdownlint-cli2
            language: system
            files: \.md$
            stages: [pre-commit]
            verbose: true
          
          # Shellcheck - FAIL on errors (block commit)
          - id: shellcheck
            name: Shell script linting
            entry: mise exec -- shellcheck
            language: system
            files: \.sh$
            stages: [pre-commit]
          
          # YAML validation - FAIL on errors
          - id: check-yaml
            name: Check YAML syntax
            entry: mise exec -- check-yaml
            language: system
            files: \.(yaml|yml)$
            stages: [pre-commit]
          
          # JSON validation - FAIL on errors
          - id: check-json
            name: Check JSON syntax
            entry: mise exec -- check-json
            language: system
            files: \.json$
            stages: [pre-commit]
          
          # Secret detection - FAIL if secrets found
          - id: detect-secrets
            name: Detect potential secrets
            entry: mise exec -- detect-secrets
            language: system
            types: [file]
            stages: [pre-commit]
    ```
    
    Note: Uses `mise exec -- tool` pattern instead of pre-commit installing its own versions. This ensures consistency between `mise run lint:*` and pre-commit hooks.
  </action>
  <verify>Run `mise exec -- pre-commit validate-config` and verify config is valid</verify>
  <done>.pre-commit-config.yaml exists with mise exec pattern hooks</done>
</task>
